安装基础工具
sudo apt update
sudo apt install -y git curl

安装 Miniconda(不要开梯子)
cd ~/Downloads

curl -O https://repo.anaconda.com/miniconda/Miniconda3-latest-Linux-x86_64.sh

bash Miniconda3-latest-Linux-x86_64.sh

创建独立虚拟环境
conda create -n agv-sim python=3.11 -y

安装第一批仿真依赖
python -m pip install --upgrade pip

python -m pip install \
    simpy \
    "networkx[default]" \
    numpy \
    pandas \
    matplotlib \
    pyyaml \
    pytest


git使用
git add .
git commit -m "说明这次改了什么"
git push

conda activate agv-sim
cd ~/projects/agv_dispatch_sim

python smoke_test.py
python smoke_test_arc_demo.py
